## 3. Block Interface Contract

Every Quest Block widget implements the same interface. This is what enables subagent-per-block parallel development.

```dart
/// Every block widget must implement this interface.
/// This is the contract between the builder and each block.
abstract class QuestBlockWidget extends StatelessWidget {
  /// The block's display metadata
  BlockMeta get meta;

  /// Whether the block is currently expanded (showing options)
  final bool isExpanded;
```

## 4. Repository Pattern

All Firestore access goes through repository classes. No direct Firestore calls from widgets or providers.

```dart
/// Base pattern for all repositories
class QuestRepository {
  final FirebaseFirestore _firestore;

  QuestRepository(this._firestore);

  CollectionReference<QuestModel> get _collection =>
      _firestore.collection('quests').withConverter<QuestModel>(
            fromFirestore: (snap, _) => QuestModel.fromJson(snap.data()!),
            toFirestore: (model, _) => model.toJson(),
          );

  /// Create
  Future<String> create(QuestModel quest) async {
    final doc = await _collection.add(quest);
    return doc.id;
  }

  /// Read single
  Future<QuestModel?> getById(String id) async {
    final snap = await _collection.doc(id).get();
    return snap.data();
  }

  /// Stream single (real-time)
  Stream<QuestModel?> watchById(String id) {
    return _collection.doc(id).snapshots().map((snap) => snap.data());
  }

  /// Query public quests by category
  Stream<List<QuestModel>> watchPublicByCategory(String category) {
    return _collection
        .where('visibility', isEqualTo: 'public')
        .where('isHidden', isEqualTo: false)
        .where('category', isEqualTo: category)
        .orderBy('addedCount', descending: true)
        .limit(50)
        .snapshots()
        .map((snap) => snap.docs.map((d) => d.data()).toList());
  }

  /// Prefix search on title
  Stream<List<QuestModel>> searchByTitle(String prefix) {
    final end = '$prefix\uf8ff';
    return _collection
        .where('visibility', isEqualTo: 'public')
        .where('isHidden', isEqualTo: false)
        .where('title', isGreaterThanOrEqualTo: prefix)
        .where('title', isLessThanOrEqualTo: end)
        .limit(20)
        .snapshots()
        .map((snap) => snap.docs.map((d) => d.data()).toList());
  }
}
```

## 8. XP Calculation Service

```dart
class XpService {
  static const int baseXp = 100;
  static const Map<Difficulty, double> difficultyMultipliers = {
    Difficulty.easy: 1.0,
    Difficulty.medium: 1.5,
    Difficulty.hard: 2.0,
    Difficulty.legendary: 3.0,
  };
  static const int firstCompletionBonus = 50;
  static const int challengeCompletionBonus = 25;
  static const int peopleBonus = 10;

  static const Map<int, int> creatorMilestones = {
    10: 10,
    50: 25,
    100: 50,
    500: 100,
  };

  /// Calculate XP for completing a non-staged quest
  static int calculateQuestXp({
    required Difficulty difficulty,
    required QuestBlocks blocks,
    required bool isFirstCompletion,
    required bool isChallenge,
    int additionalPeople = 0,
  }) {
    double xp = baseXp * difficultyMultipliers[difficulty]!;

    if (isFirstCompletion) xp += firstCompletionBonus;
    if (isChallenge) xp += challengeCompletionBonus;
    if (blocks.people != null && additionalPeople > 0) {
      xp += additionalPeople * peopleBonus;
    }
    if (blocks.bonus != null) {
      // Bonus condition evaluation handled separately
      xp += blocks.bonus!.xpBonus;
    }

    return xp.round();
  }

  /// Calculate XP for completing a single stage
  static int calculateStageXp({
    required int stageBaseXp,
    required Difficulty difficulty,
  }) {
    return (stageBaseXp * difficultyMultipliers[difficulty]!).round();
  }

  /// Determine tier from total XP
  static String tierFromXp(int xp) {
    if (xp >= 15000) return 'legend';
    if (xp >= 5000) return 'trailblazer';
    if (xp >= 2000) return 'adventurer';
    if (xp >= 500) return 'explorer';
    return 'novice';
  }
}
```

## 9. Firestore Security Rules

```
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // --- Users ---
    match /users/{userId} {
      allow read: if request.auth != null;
      allow create: if request.auth.uid == userId;
      allow update: if request.auth.uid == userId;
      allow delete: if request.auth.uid == userId;

      // UserQuests subcollection
      match /quests/{userQuestId} {
        allow read, write: if request.auth.uid == userId;
      }
    }

    // --- Quests ---
    match /quests/{questId} {
      allow read: if request.auth != null
        && (resource.data.visibility == 'public' || resource.data.creatorId == request.auth.uid);
      allow create: if request.auth != null
        && request.resource.data.creatorId == request.auth.uid;
      allow update: if request.auth != null
        && resource.data.creatorId == request.auth.uid;
      allow delete: if request.auth != null
        && resource.data.creatorId == request.auth.uid;
    }

    // --- Friendships ---
    match /friendships/{friendshipId} {
      allow read: if request.auth != null
        && request.auth.uid in resource.data.userIds;
      allow create: if request.auth != null
        && request.auth.uid == request.resource.data.requesterId
        && request.auth.uid in request.resource.data.userIds;
      allow update: if request.auth != null
        && request.auth.uid in resource.data.userIds
        && request.auth.uid != resource.data.requesterId;
      allow delete: if request.auth != null
        && request.auth.uid in resource.data.userIds;
    }

    // --- Challenges ---
    match /challenges/{challengeId} {
      allow read: if request.auth != null
        && (resource.data.senderId == request.auth.uid
          || resource.data.receiverId == request.auth.uid);
      allow create: if request.auth != null
        && request.resource.data.senderId == request.auth.uid;
      allow update: if request.auth != null
        && resource.data.receiverId == request.auth.uid;
    }

    // --- Activity ---
    match /activity/{activityId} {
      allow read: if request.auth != null;
      // Only Cloud Functions write activity
      allow write: if false;
    }

    // --- Reports ---
    match /reports/{reportId} {
      allow create: if request.auth != null
        && request.resource.data.reporterId == request.auth.uid;
      // Only moderators can read/update
      allow read, update: if false;
    }

    // --- Quality Signals ---
    match /qualitySignals/{signalId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null
        && request.resource.data.userId == request.auth.uid;
      // Write-once: no updates or deletes
      allow update, delete: if false;
    }

    // --- Leaderboards (precomputed by Cloud Functions) ---
    match /leaderboards/{leaderboardId} {
      allow read: if request.auth != null;
      allow write: if false;
    }

    // --- Deleted Usernames ---
    match /deletedUsernames/{username} {
      allow read: if request.auth != null;
      allow write: if false;
    }
  }
```
